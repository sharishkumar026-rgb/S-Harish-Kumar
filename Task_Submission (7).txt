Task - 3 Multiple Row Subquery

1. Using ANY

</> SQL

SELECT *
FROM studentmark
WHERE marks > ANY (
    SELECT marks 
    FROM studentmark 
    WHERE department = 'ECE'
);

2. Using ALL

SELECT *
FROM studentmark
WHERE marks > ALL (
    SELECT marks 
    FROM studentmark 
    WHERE department = 'ECE'
);

Task - 4 Correlated Subquery

SELECT *
FROM studentmark s1
WHERE marks > (
    SELECT AVG(marks)
    FROM studentmark s2
    WHERE s1.department = s2.department
);
